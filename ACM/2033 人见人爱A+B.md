```
#include<stdio.h>
int main() {
	int  m, n, a[6];
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) {
		for (int j = 0; j < 6; j++) {
			scanf("%d", &a[j]);
		}
		int b[3] = { 0 };
		for (int k = 2; k >= 0; k--) {
			b[k] =b[k]+ a[k] + a[k + 3];
			while (b[k] > 59 && k != 0) {
				b[k] -= 60;
				b[k - 1] += 1;
			}
		}
		for (int z = 0; z < 3; z++) {
			if (z == 2)
				printf("%d", b[z]);
			else {
				printf("%d ", b[z]);
			}
		}
		printf("\n");
	}
	return 0;
}
```
