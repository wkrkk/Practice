#include<stdio.h>
int main() {
	int n, m, temp, a[100],b[100];
	while (scanf("%d%d", &n, &m)) {
		if (n == 0 && m == 0)
			break;
		else {
			for (int i = 0; i < n; i++)
				scanf("%d", &a[i]);
			for (int j = 0; j < m; j++)
				scanf("%d", &b[j]);
			for (int i = 1; i < n; i++) {
				for (int j = 0; j < n - i; j++) {
					if (a[j] > a[j + 1])
					{
						temp = a[j];
						a[j] = a[j + 1];
						a[j + 1] = temp;
					}
				}
			}
			if (n == 0) 
				printf("NULL\n");
			else if (m == 0) {
				for (int i = 0; i < n; i++)
					printf("%d ", a[i]);
				printf("\n");
			}
			else {
				int count = 0;
				for (int i = 0; i < n; i++) {
					for (int j = 0; j < m; j++) {
						if (a[i] - b[j] == 0) {
							a[i] = 0;
							count++;
							continue;
						}
					}
				}
				for (int i = 0; i < n; i++) {
					if (count == n) {
						printf("NULL");
						break;
					}
					else {
						if (a[i] != 0)
							printf("%d ", a[i]);
					}
				}
				printf("\n");
			}

		}
	}
	return 0;
}
