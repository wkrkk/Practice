### 2020 绝对值排序

```
#include<stdio.h>
#include<math.h>
void rank(int a[],int n) {
	int temp;
	for (int i = 1; i < n; i++) {
		for (int j = 0; j < n - i; j++) {
			if (abs(a[j]) < abs(a[j + 1]))
			{
				temp = a[j];
				a[j] = a[j + 1];
				a[j + 1] = temp;
			}
		}
	}
	for (int i = 0; i < n ; i++) {
		if (i == n - 1)
			printf("%d", a[i]);
		else
		    printf("%d ", a[i]);
	}
}
int main() {
	int n, a[100];
	while (scanf("%d", &n)){
		if (n == 0)
			break;
		else {
			for (int i = 0; i < n; i++)
				scanf("%d", &a[i]);
			rank(a, n);
			printf("\n");
		}
	}
	return 0;
}
```
