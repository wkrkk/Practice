```
#include<stdio.h>
int main() {
	int a, b, c, d, m;
	for (m = 1000; m < 10000; m++) {
		d = m % 10;
		c = m / 10 % 10;
		b = m / 100 % 10;
		a = m / 1000 % 10;
		if (a == d && b == c)
			printf("%d\n", m);	
	}
	return 0;
}
```
