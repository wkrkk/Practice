#include<stdio.h>
int main() {
	int i, n, ans = 1;
	scanf("%d", &n);
	for (i = 1; i <= n; i++) {
		ans *= i;
		while (ans % 10 == 0) 
			ans /= 10;//去除末尾的0
		ans %= 1000;//取余时取大，防止进位出错
	}
	ans %= 10;//直接取最右边非0数字
	printf("%d", ans);
	return 0;
}
