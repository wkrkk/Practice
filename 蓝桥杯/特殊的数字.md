### 特殊的数字

![](https://github.com/wkrkk/RandomPictures/blob/master/TIM%E6%88%AA%E5%9B%BE20190225105545.png?raw=true)

```
#include<stdio.h>
int main() {
	int a, b, c, sum, sum1;
	
	for (sum = 100; sum < 1000;sum++) {
		c = sum % 10;
		b = sum / 10 % 10;
		a = sum / 100 % 10;
		sum1 = a * a * a + b * b * b + c * c * c;
		if (sum == sum1)
			printf("%d\n", sum);
	}
	return 0;
}
```
