### 杨辉三角形

![](https://github.com/wkrkk/RandomPictures/blob/master/TIM%E6%88%AA%E5%9B%BE20190225195922.png?raw=true)

```
#include<stdio.h>
int main() {
	int n, i, j, a[34][34];
	scanf("%d", &n);
	for(i=0;i<n;i++)
		for (j = 0; j <= i; j++) {
			if (j == 0 || j == i)
				a[i][j] = 1;
			else {
				a[i][j] = a[i - 1][j - 1] + a[i - 1][j];
			}
		}
	for(i=0;i<n;i++)
		for (j = 0; j <= i; j++) {
			if (j == i)
				printf("%d\n", a[i][j]);
			else
				printf("%d ", a[i][j]);		
		}
	return 0;
}
```
