#### 山羊拉丁文

[题目](https://leetcode-cn.com/problems/goat-latin)

```
class Solution(object):
    def toGoatLatin(self,s):
        letter=['a','e','i','o','u','A','E','I','O','U']
        words=s.split() #分割字符串
        s1=''  #设置新的字符串
        for i,word in enumerate(words):
            if word[0] in letter:
                s1+=word+"ma"+'a'*(i+1)+' '
            else:
                s1+=word[1:]+word[0]+"ma"+'a'*(i+1)+' '
        return s1[:-1]

s=input()
solution=Solution()
s1=solution.toGoatLatin(s)
print(s1)
```
